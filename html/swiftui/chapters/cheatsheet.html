Шпаргалка: Swift vs JavaScript

Быстрая справка для разработчиков, переходящих с JS/PHP на Swift

1 Переменные и константы

JavaScript

let name = "John"       // можно изменить
const age = 25          // константа
var old = "устарело"    // не используйте

Swift

var name = "John"       // можно изменить
let age = 25            // константа (НАОБОРОТ!)
// Предпочитайте let когда возможно
Внимание: В Swift let = константа, var = переменная. Наоборот от JS!

3 Optionals (nil-безопасность) 

// Объявление Optional
var name: String? = nil        // Может быть nil
var age: Int = 25              // НЕ может быть nil

// Безопасное извлечение
if let unwrapped = name {
    print(unwrapped)           // Есть значение
}

// Optional chaining
let length = name?.count       // nil если name = nil

// Значение по умолчанию
let displayName = name ?? "Гость"

// Force unwrap (опасно!)
let forced = name!             // Крэш если nil

4 Функции

JavaScript

// Обычная функция
function greet(name) {
    return "Hello, " + name
}

// Arrow function
const greet = (name) => "Hello, " + name

// Async
async function fetchData() {
    const data = await fetch(url)
    return data.json()
}

Swift

// Обычная функция
func greet(name: String) -> String {
    return "Hello, \(name)"
}

// Closure (как arrow function)
let greet = { (name: String) -> String in
    "Hello, \(name)"
}

// Async
func fetchData() async throws -> Data {
    let data = try await URLSession.shared.data(from: url)
    return data
}

5 Работа с коллекциями 

let numbers = [1, 2, 3, 4, 5]

// Map - преобразование
let doubled = numbers.map { $0 * 2 }        // [2, 4, 6, 8, 10]

// Filter - фильтрация
let even = numbers.filter { $0 % 2 == 0 }   // [2, 4]

// Reduce - свёртка
let sum = numbers.reduce(0, +)              // 15

// ForEach - перебор
numbers.forEach { print($0) }

// First/Last
let first = numbers.first                    // Optional(1)
let last = numbers.last                      // Optional(5)

// Contains
let hasThree = numbers.contains(3)           // true

// Sorted
let sorted = numbers.sorted(by: >)           // [5, 4, 3, 2, 1]

$0, $1, $2...

Сокращённые имена параметров в замыканиях. $0 = первый параметр, $1 = второй и т.д.

6 Условия и циклы

If / Else

if age >= 18 {
    print("Взрослый")
} else if age >= 13 {
    print("Подросток")
} else {
    print("Ребёнок")
}

// Без скобок вокруг условия!

Switch

switch status {
case "active":
    print("Активен")
case "pending", "review":
    print("В процессе")
default:
    print("Неизвестно")
}
// Не нужен break!

For Loop

// По диапазону
for i in 0..<5 { }      // 0,1,2,3,4
for i in 0...5 { }      // 0,1,2,3,4,5

// По массиву
for item in items { }

// С индексом
for (i, item) in items.enumerated() { }

Guard

// Early return
guard let user = user else {
    return  // Выход если nil
}
// user теперь не Optional!

guard age >= 18 else {
    throw Error.underage
}

7 Struct vs Class

Struct (Value Type)

struct Product {
    var name: String
    var price: Double

    func formatted() -> String {
        "\(name): $\(price)"
    }
}

var a = Product(name: "iPhone", price: 999)
var b = a           // Копия!
b.name = "iPad"
print(a.name)       // "iPhone" (не изменился)

Используйте для данных: Product, User, Order

Class (Reference Type)

class CartManager {
    var items: [Product] = []

    func add(_ product: Product) {
        items.append(product)
    }
}

let a = CartManager()
let b = a           // Ссылка!
b.items.append(product)
print(a.items.count) // Изменился!

Используйте для сервисов: Manager, ViewModel

8 SwiftUI Property Wrappers

Wrapper Назначение Аналог в React
@State Локальное состояние View useState()
@Binding Связь с родительским @State props + onChange
@Observable Класс с реактивными свойствами MobX observable
@Environment Глобальные значения (тема, locale) useContext()
@AppStorage Сохранение в UserDefaults localStorage + useState 
struct CounterView: View {
    @State private var count = 0          // Локальное состояние

    var body: some View {
        Button("Count: \(count)") {
            count += 1                     // UI обновится автоматически
        }
    }
}

@Observable
class CartViewModel {
    var items: [Product] = []             // Реактивное свойство
    var total: Double { items.reduce(0) { $0 + $1.price } }
}

10 Обработка ошибок

JavaScript

try {
    const data = await fetch(url)
    const json = await data.json()
} catch (error) {
    console.error(error)
}

Swift

do {
    let data = try await URLSession.shared.data(from: url)
    let products = try JSONDecoder().decode([Product].self, from: data)
} catch {
    print(error)
}

// Или с optional result
let data = try? await fetchData()   // nil если ошибка

// Игнорировать ошибку (опасно)
let data = try! await fetchData()   // Крэш если ошибка
