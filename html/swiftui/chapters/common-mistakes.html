Оглавление Частые ошибки

Частые ошибки новичков

Типичные ошибки при переходе с JavaScript/PHP на Swift и SwiftUI

1

Путаница let и var

В Swift let и var работают НАОБОРОТ от JavaScript

❌

Неправильно

// Думаете как в JS: let = можно изменить
let count = 0
count = 1  // Ошибка! Cannot assign to value: 'count' is a 'let' constant
✅

Правильно

// В Swift: var = можно изменить, let = константа
var count = 0
count = 1  // OK!

let maxCount = 100  // Константа, нельзя изменить

Запомните:

Swift: let = const, var = let (в JS)

2

Force Unwrap везде

Использование ! для распаковки Optional приводит к крэшам

❌

Опасно

let user: User? = nil
print(user!.name)  // КРЭШ! Fatal error: Unexpectedly found nil

let text: String? = textField.text
let length = text!.count  // Крэш если пользователь ничего не ввёл
✅

Безопасно

// Вариант 1: if let
if let user = user {
    print(user.name)
}

// Вариант 2: guard let (early return)
guard let user = user else { return }
print(user.name)

// Вариант 3: значение по умолчанию
let length = text?.count ?? 0

// Вариант 4: optional chaining
let name = user?.profile?.name  // nil если что-то в цепочке nil
3

Забыли ? у Optional

Не указали что значение может быть nil

❌

Ошибка компиляции

struct User {
    var name: String
    var email: String  // Что если email не указан?
}

// Ошибка: нужно передать ВСЕ поля
let user = User(name: "John")  // Missing argument for parameter 'email'
✅

Правильно

struct User {
    var name: String
    var email: String?  // Может быть nil
}

// Теперь можно создать без email
let user = User(name: "John", email: nil)
// или
let user2 = User(name: "John", email: "john@example.com")
4

Изменение состояния без @State

View в SwiftUI - это struct, его нельзя просто изменить

❌

Ошибка компиляции

struct CounterView: View {
    var count = 0  // Просто свойство

    var body: some View {
        Button("Count: \(count)") {
            count += 1  // Ошибка! Cannot assign to property: 'self' is immutable
        }
    }
}
✅

Правильно

struct CounterView: View {
    @State private var count = 0  // Добавили @State!

    var body: some View {
        Button("Count: \(count)") {
            count += 1  // Теперь работает!
        }
    }
}

Правило:

Если переменная должна изменяться и вызывать перерисовку UI - используйте @State

5

Неправильный синтаксис Binding

Забыли $ при передаче @Binding

❌

Ошибка

struct ParentView: View {
    @State private var text = ""

    var body: some View {
        // Передаём значение, не binding
        ChildView(text: text)  // Ошибка! Cannot convert value of type 'String' to expected argument type 'Binding<String>'
    }
}

struct ChildView: View {
    @Binding var text: String
    // ...
}
✅

Правильно

struct ParentView: View {
    @State private var text = ""

    var body: some View {
        // $ создаёт binding к @State
        ChildView(text: $text)  // Правильно!

        // То же самое для стандартных компонентов
        TextField("Введите текст", text: $text)
        Toggle("Включить", isOn: $isEnabled)
    }
}
6

Асинхронный код в body

Попытка вызвать async функцию напрямую в body

❌

Неправильно

struct ProductsView: View {
    var body: some View {
        VStack {
            // Нельзя вызывать async напрямую
            let products = await fetchProducts()  // Ошибка!
        }
    }
}
✅

Правильно

struct ProductsView: View {
    @State private var products: [Product] = []

    var body: some View {
        List(products) { product in
            Text(product.title)
        }
        .task {  // Специальный модификатор для async
            products = await fetchProducts()
        }
        // или
        .onAppear {
            Task {
                products = await fetchProducts()
            }
        }
    }
}
7

Сравнение Optional без распаковки

Попытка использовать Optional как обычное значение

❌

Ошибка

let count: Int? = 5

// Ошибка: нельзя сравнить Optional и Int напрямую
if count > 0 {  // Value of optional type 'Int?' must be unwrapped
    print("Positive")
}
✅

Правильно

let count: Int? = 5

// Вариант 1: if let
if let count = count, count > 0 {
    print("Positive")
}

// Вариант 2: nil coalescing
if (count ?? 0) > 0 {
    print("Positive")
}

// Вариант 3: сравнение с Optional напрямую
if count == 5 {  // Это работает!
    print("Five")
}
8

Забыли return в многострочной функции

В однострочных функциях return не нужен, в многострочных - обязателен

❌

Ошибка

func calculateTotal(items: [Product]) -> Double {
    let sum = items.reduce(0) { $0 + $1.price }
    sum  // Ошибка! Missing return
}
✅

Правильно

// Многострочная - нужен return
func calculateTotal(items: [Product]) -> Double {
    let sum = items.reduce(0) { $0 + $1.price }
    return sum
}

// Однострочная - return не нужен
func calculateTotal(items: [Product]) -> Double {
    items.reduce(0) { $0 + $1.price }
}
9

Смешанные типы в массиве

В Swift массив может содержать только один тип данных

❌

Как в JavaScript (не работает)

// В JS можно: [1, "два", true, {name: "John"}]
// В Swift нельзя:
let mixed = [1, "два", true]  // Ошибка! Heterogeneous collection literal
✅

Правильно в Swift

// Один тип
let numbers: [Int] = [1, 2, 3]
let names: [String] = ["John", "Jane"]

// Если ОЧЕНЬ нужны разные типы (редко!)
let mixed: [Any] = [1, "два", true]

// Лучше использовать struct
struct Item {
    var id: Int
    var name: String
    var active: Bool
}
10

Когда использовать @State vs @Observable

Выбор между локальным состоянием и ViewModel

Ситуация Используйте
Простое UI состояние (открыт ли popup, выбрана ли вкладка) @State
Состояние формы (текст input, выбранная дата) @State
Данные из API (список товаров, профиль пользователя) @Observable ViewModel
Бизнес-логика (корзина, избранное, авторизация) @Observable ViewModel
Состояние, которое нужно в нескольких View @Observable + @Environment 
// @State - для простого локального UI
struct ProductCard: View {
    @State private var isExpanded = false  // Локальное состояние

    var body: some View {
        // ...
    }
}

// @Observable - для бизнес-логики и данных
@Observable
class ProductsViewModel {
    var products: [Product] = []
    var isLoading = false

    func loadProducts() async {
        isLoading = true
        products = try await api.fetchProducts()
        isLoading = false
    }
}

Краткий чек-лист

  • ✓ let = константа, var = переменная (наоборот от JS!)
  • ✓ Избегайте ! - используйте if let, guard let, ??
  • ✓ Добавляйте ? к типу если значение может быть nil
  • ✓ Для изменяемого состояния View используйте @State
  • ✓ При передаче @Binding используйте $
  • ✓ Для async используйте .task { } или Task { }
  • ✓ Массивы могут содержать только один тип
