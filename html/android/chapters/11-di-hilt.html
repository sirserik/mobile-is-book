Глава 11: Hilt Dependency Injection

Что такое Dependency Injection?

Dependency Injection (DI) — это паттерн, при котором зависимости передаются объекту извне, а не создаются внутри него. Это упрощает тестирование и делает код гибким.

Без DI (плохо)

class HomeViewModel : ViewModel() {
    // Жёсткая связь — сложно тестировать
    private val repo = ProductRepositoryImpl(
        api = Retrofit.create(ProductApi::class)
    )
}

С DI (хорошо)

@HiltViewModel
class HomeViewModel @Inject constructor(
    // Зависимость передаётся извне
    private val repo: ProductRepository
) : ViewModel()

Преимущества DI

  • • Тестируемость — легко подменить зависимости на mock/fake
  • • Гибкость — можно менять реализации без изменения кода
  • • Модульность — чёткое разделение ответственности
  • • Переиспользуемость — один экземпляр на всё приложение

Hilt Modules

Модули говорят Hilt как создавать зависимости. Есть два способа: @Provides и @Binds.

@Provides — для внешних классов

Используйте для классов, которые вы не можете изменить (Retrofit, Room, OkHttp). 

@Module
@InstallIn(SingletonComponent::class)
object NetworkModule {

    @Provides
    @Singleton
    fun provideMoshi(): Moshi {
        return Moshi.Builder()
            .add(KotlinJsonAdapterFactory())
            .build()
    }

    @Provides
    @Singleton
    fun provideOkHttpClient(): OkHttpClient {
        return OkHttpClient.Builder()
            .addInterceptor(HttpLoggingInterceptor().apply {
                level = HttpLoggingInterceptor.Level.BODY
            })
            .connectTimeout(30, TimeUnit.SECONDS)
            .readTimeout(30, TimeUnit.SECONDS)
            .build()
    }

    @Provides
    @Singleton
    fun provideRetrofit(
        okHttpClient: OkHttpClient,
        moshi: Moshi
    ): Retrofit {
        return Retrofit.Builder()
            .baseUrl("https://dummyjson.com/")
            .client(okHttpClient)
            .addConverterFactory(MoshiConverterFactory.create(moshi))
            .build()
    }

    @Provides
    @Singleton
    fun provideProductApi(retrofit: Retrofit): ProductApi {
        return retrofit.create(ProductApi::class.java)
    }
}

@Binds — для интерфейсов

Используйте для связывания интерфейса с его реализацией. 

@Module
@InstallIn(SingletonComponent::class)
abstract class RepositoryModule {

    @Binds
    @Singleton
    abstract fun bindProductRepository(
        impl: ProductRepositoryImpl
    ): ProductRepository

    @Binds
    @Singleton
    abstract fun bindCartRepository(
        impl: CartRepositoryImpl
    ): CartRepository

    @Binds
    @Singleton
    abstract fun bindFavoriteRepository(
        impl: FavoriteRepositoryImpl
    ): FavoriteRepository
}

Database Module

@Module
@InstallIn(SingletonComponent::class)
object DatabaseModule {

    @Provides
    @Singleton
    fun provideDatabase(
        @ApplicationContext context: Context
    ): EcommerceDatabase {
        return Room.databaseBuilder(
            context,
            EcommerceDatabase::class.java,
            "ecommerce.db"
        )
        .fallbackToDestructiveMigration()
        .build()
    }

    @Provides
    fun provideProductDao(db: EcommerceDatabase): ProductDao = db.productDao()

    @Provides
    fun provideCartDao(db: EcommerceDatabase): CartDao = db.cartDao()

    @Provides
    fun provideFavoriteDao(db: EcommerceDatabase): FavoriteDao = db.favoriteDao()
}

UseCase с @Inject

UseCase не требуют модуля — достаточно @Inject constructor. 

class GetProductsUseCase @Inject constructor(
    private val repository: ProductRepository
) {
    operator fun invoke(): Flow<Resource<List<Product>>> {
        return repository.getProducts()
    }
}

class GetProductByIdUseCase @Inject constructor(
    private val repository: ProductRepository
) {
    operator fun invoke(id: Int): Flow<Resource<Product>> {
        return repository.getProductById(id)
    }
}

class AddToCartUseCase @Inject constructor(
    private val cartRepository: CartRepository
) {
    suspend operator fun invoke(product: Product) {
        cartRepository.addToCart(product)
    }
}

Qualifiers — различение зависимостей

Когда есть несколько зависимостей одного типа, используйте Qualifier. 

// Создаём Qualifier аннотации
@Qualifier
@Retention(AnnotationRetention.BINARY)
annotation class IoDispatcher

@Qualifier
@Retention(AnnotationRetention.BINARY)
annotation class MainDispatcher

@Qualifier
@Retention(AnnotationRetention.BINARY)
annotation class DefaultDispatcher

// В модуле
@Module
@InstallIn(SingletonComponent::class)
object DispatcherModule {

    @Provides
    @IoDispatcher
    fun provideIoDispatcher(): CoroutineDispatcher = Dispatchers.IO

    @Provides
    @MainDispatcher
    fun provideMainDispatcher(): CoroutineDispatcher = Dispatchers.Main

    @Provides
    @DefaultDispatcher
    fun provideDefaultDispatcher(): CoroutineDispatcher = Dispatchers.Default
}

// Использование
class ProductRepositoryImpl @Inject constructor(
    private val api: ProductApi,
    @IoDispatcher private val ioDispatcher: CoroutineDispatcher
) : ProductRepository {

    override fun getProducts() = flow {
        val products = api.getProducts()
        emit(products)
    }.flowOn(ioDispatcher)
}

Тестирование с Hilt

// Замена реальной зависимости на fake для тестов
@Module
@TestInstallIn(
    components = [SingletonComponent::class],
    replaces = [RepositoryModule::class]
)
abstract class FakeRepositoryModule {

    @Binds
    @Singleton
    abstract fun bindProductRepository(
        impl: FakeProductRepository
    ): ProductRepository
}

// Fake реализация для тестов
class FakeProductRepository @Inject constructor() : ProductRepository {

    private val fakeProducts = listOf(
        Product(id = 1, title = "Test iPhone", price = 999.0, ...),
        Product(id = 2, title = "Test MacBook", price = 1999.0, ...)
    )

    override fun getProducts() = flow {
        emit(Resource.Success(fakeProducts))
    }

    override fun getProductById(id: Int) = flow {
        val product = fakeProducts.find { it.id == id }
        if (product != null) {
            emit(Resource.Success(product))
        } else {
            emit(Resource.Error("Product not found"))
        }
    }
}

Частые ошибки

Забыли @AndroidEntryPoint

Activity/Fragment без этой аннотации не получит зависимости.

@HiltViewModel без @Inject constructor

ViewModel требует обе аннотации для работы.

@Singleton без @InstallIn(SingletonComponent)

Scope должен соответствовать компоненту.

Ключевые концепции

  • • @HiltAndroidApp — точка входа для Hilt в Application
  • • @AndroidEntryPoint — включает DI для Activity/Fragment
  • • @HiltViewModel — DI для ViewModel
  • • @Module + @InstallIn — модули с зависимостями
  • • @Provides — для внешних классов (Retrofit, Room)
  • • @Binds — для связывания интерфейсов
  • • @Singleton — один экземпляр на приложение
  • • @Qualifier — различение зависимостей одного типа
