Kotlin для JS/PHP разработчиков

Быстрый переход с JavaScript/TypeScript/PHP на Kotlin

1. Переменные

JavaScript / TypeScript

let name = "John"      // изменяемая
const age = 25         // константа
let score: number = 100 // с типом

Kotlin

var name = "John"      // изменяемая
val age = 25           // константа (immutable)
var score: Int = 100  // с типом

Правило: Всегда используйте val по умолчанию. Используйте var только когда значение должно меняться.

2. Nullable типы (важно!)

JavaScript / TypeScript

let name: string | null = null
let user: User | undefined

// Optional chaining
const city = user?.address?.city

// Nullish coalescing
const result = value ?? "default"

Kotlin

var name: String? = null  // ? = nullable
var user: User? = null

// Safe call
val city = user?.address?.city

// Elvis operator
val result = value ?: "default"

Опасно: !! — force unwrap. Выбросит исключение если null. Избегайте! 

val name = user!!.name  // NPE если user == null

3. Функции

JavaScript / TypeScript

// Обычная функция
function greet(name: string): string {
    return `Hello, ${name}!`
}

// Arrow function
const add = (a: number, b: number) => a + b

// Default параметр
function log(msg: string, level = "info") {}

Kotlin

// Обычная функция
fun greet(name: String): String {
    return "Hello, $name!"
}

// Single expression
fun add(a: Int, b: Int) = a + b

// Default параметр
fun log(msg: String, level: String = "info") {}

4. Data Class (аналог interface/type)

TypeScript

interface User {
    id: number
    name: string
    email: string
}

const user: User = {
    id: 1,
    name: "John",
    email: "john@mail.com"
}

Kotlin

data class User(
    val id: Int,
    val name: String,
    val email: String
)

val user = User(
    id = 1,
    name = "John",
    email = "john@mail.com"
)

Бонус: data class автоматически генерирует equals(), hashCode(), toString(), copy()

5. Коллекции

JavaScript

const nums = [1, 2, 3]
const doubled = nums.map(n => n * 2)
const evens = nums.filter(n => n % 2 === 0)
const sum = nums.reduce((a, b) => a + b, 0)
const first = nums.find(n => n > 1)

Kotlin

val nums = listOf(1, 2, 3)
val doubled = nums.map { it * 2 }
val evens = nums.filter { it % 2 == 0 }
val sum = nums.reduce { a, b -> a + b }
val first = nums.find { it > 1 }

6. Асинхронность: Coroutines

JavaScript async/await

async function fetchUser(): Promise<User> {
    const response = await fetch(url)
    return response.json()
}

// Вызов
const user = await fetchUser()

Kotlin Coroutines

suspend fun fetchUser(): User {
    val response = api.getUser()
    return response
}

// Вызов
viewModelScope.launch {
    val user = fetchUser()
}

Ключевое: suspend функции можно вызывать только из других suspend функций или coroutine scope.

7. Лямбды и замыкания

JavaScript

// Arrow function
const onClick = (item) => {
    console.log(item)
}

// Callback
button.addEventListener('click', () => {
    console.log('clicked')
})

Kotlin

// Lambda
val onClick: (Item) -> Unit = { item ->
    println(item)
}

// Trailing lambda
button.setOnClickListener {
    println("clicked")
}

8. When (switch на стероидах)

JavaScript

switch (status) {
    case 'loading':
        return 'Loading...'
    case 'success':
        return 'Done!'
    case 'error':
        return 'Failed'
    default:
        return 'Unknown'
}

Kotlin

when (status) {
    "loading" -> "Loading..."
    "success" -> "Done!"
    "error" -> "Failed"
    else -> "Unknown"
}

// С типами (sealed class)
when (state) {
    is Loading -> showLoader()
    is Success -> showData(state.data)
    is Error -> showError(state.message)
}

Шпаргалка: JS → Kotlin с примерами

let → var — изменяемая переменная 
// JavaScript
let count = 0
count = 1  // OK
// Kotlin
var count = 0
count = 1  // OK
const → val — неизменяемая (рекомендуется по умолчанию) 
// JavaScript
const name = "John"
name = "Bob"  // Ошибка!
// Kotlin
val name = "John"
name = "Bob"  // Ошибка!
string | null → String? — может быть null 
// TypeScript
let name: string | null = null
let user: User | undefined
// Kotlin — ? означает nullable
var name: String? = null
var user: User? = null
?? → ?: — значение по умолчанию если null 
// JavaScript — nullish coalescing
const result = value ?? "default"
// Если value = null/undefined → "default"
// Kotlin — Elvis operator
val result = value ?: "default"
// Если value = null → "default"
Promise → suspend / Flow — асинхронные операции

Promise — одно значение в будущем (запрос к API).
suspend — функция которая может "приостановиться" (как await).
Flow — поток значений (как Observable в RxJS). 

// JavaScript — одно значение
async function getUser(): Promise<User> {
  const res = await fetch(url)
  return res.json()
}

// Вызов
const user = await getUser()
// Kotlin — одно значение
suspend fun getUser(): User {
  return api.getUser()
}

// Вызов из coroutine
val user = getUser()
// JavaScript — поток значений
const observable = new Observable(...)
observable.subscribe(value => {
  console.log(value)
})
// Kotlin — Flow (поток значений)
fun getProducts(): Flow<List<Product>>

// Подписка
getProducts().collect { products ->
  println(products)
}
async/await → suspend — ожидание без блокировки 
// JavaScript
async function loadData() {
  const a = await fetchA()
  const b = await fetchB()
  return { a, b }
}
// Kotlin
suspend fun loadData(): Data {
  val a = fetchA()  // await неявный
  val b = fetchB()
  return Data(a, b)
}
interface/type → data class — модель данных 
// TypeScript
interface User {
  id: number
  name: string
}
const user: User = { id: 1, name: "John" }
// Kotlin
data class User(
  val id: Int,
  val name: String
)
val user = User(id = 1, name = "John")
(x) => x * 2 → { x -> x * 2 } — лямбда-функция 
// JavaScript
const doubled = nums.map(n => n * 2)
const evens = nums.filter(n => n % 2 === 0)
button.onClick(() => alert("Hi"))
// Kotlin — it = единственный параметр
val doubled = nums.map { it * 2 }
val evens = nums.filter { it % 2 == 0 }
button.onClick { showAlert("Hi") }
switch → when — множественный выбор 
// JavaScript
switch (status) {
  case "loading": return "..."
  case "success": return "OK"
  default: return "?"
}
// Kotlin — без break, возвращает значение
when (status) {
  "loading" -> "..."
  "success" -> "OK"
  else -> "?"
}
