Определение маршрутов

sealed class Screen(val route: String) {
    object Home : Screen("home")
    object Search : Screen("search")
    object Favorites : Screen("favorites")
    object Cart : Screen("cart")
    object Profile : Screen("profile")

    object ProductDetail : Screen("product/{productId}") {
        fun createRoute(productId: Int) = "product/$productId"
    }

    object Category : Screen("category/{categoryName}") {
        fun createRoute(name: String) = "category/$name"
    }
}

NavGraph

@Composable
fun EcommerceNavGraph(
    navController: NavHostController = rememberNavController()
) {
    NavHost(
        navController = navController,
        startDestination = Screen.Home.route
    ) {
        // Главная
        composable(route = Screen.Home.route) {
            HomeScreen(
                onProductClick = { productId ->
                    navController.navigate(Screen.ProductDetail.createRoute(productId))
                },
                onSearchClick = {
                    navController.navigate(Screen.Search.route)
                }
            )
        }

        // Детали товара
        composable(
            route = Screen.ProductDetail.route,
            arguments = listOf(
                navArgument("productId") { type = NavType.IntType }
            )
        ) {
            DetailScreen(
                onBackClick = { navController.popBackStack() }
            )
        }

        // Поиск
        composable(route = Screen.Search.route) {
            SearchScreen(
                onProductClick = { productId ->
                    navController.navigate(Screen.ProductDetail.createRoute(productId))
                },
                onBackClick = { navController.popBackStack() }
            )
        }

        // Избранное
        composable(route = Screen.Favorites.route) {
            FavoritesScreen(
                onProductClick = { productId ->
                    navController.navigate(Screen.ProductDetail.createRoute(productId))
                }
            )
        }

        // Корзина
        composable(route = Screen.Cart.route) {
            CartScreen()
        }
    }
}

Bottom Navigation

sealed class BottomNavItem(
    val route: String,
    val icon: ImageVector,
    val label: String
) {
    object Home : BottomNavItem("home", Icons.Default.Home, "Главная")
    object Favorites : BottomNavItem("favorites", Icons.Default.Favorite, "Избранное")
    object Cart : BottomNavItem("cart", Icons.Default.ShoppingCart, "Корзина")
    object Profile : BottomNavItem("profile", Icons.Default.Person, "Профиль")
}

@Composable
fun BottomNavigationBar(
    navController: NavHostController,
    cartCount: Int
) {
    val items = listOf(
        BottomNavItem.Home,
        BottomNavItem.Favorites,
        BottomNavItem.Cart,
        BottomNavItem.Profile
    )

    val navBackStackEntry by navController.currentBackStackEntryAsState()
    val currentRoute = navBackStackEntry?.destination?.route

    NavigationBar {
        items.forEach { item ->
            NavigationBarItem(
                selected = currentRoute == item.route,
                onClick = {
                    navController.navigate(item.route) {
                        // Избегаем дублирования в стеке
                        popUpTo(navController.graph.startDestinationId) {
                            saveState = true
                        }
                        launchSingleTop = true
                        restoreState = true
                    }
                },
                icon = {
                    if (item == BottomNavItem.Cart && cartCount > 0) {
                        BadgedBox(badge = { Badge { Text(cartCount.toString()) } }) {
                            Icon(item.icon, item.label)
                        }
                    } else {
                        Icon(item.icon, item.label)
                    }
                },
                label = { Text(item.label) }
            )
        }
    }
}

MainScreen с Navigation

@Composable
fun MainScreen() {
    val navController = rememberNavController()
    val cartViewModel: CartViewModel = hiltViewModel()
    val cartCount by cartViewModel.cartCount.collectAsStateWithLifecycle()

    val navBackStackEntry by navController.currentBackStackEntryAsState()
    val currentRoute = navBackStackEntry?.destination?.route

    // Скрываем bottom bar на экране деталей
    val showBottomBar = currentRoute in listOf(
        Screen.Home.route,
        Screen.Favorites.route,
        Screen.Cart.route,
        Screen.Profile.route
    )

    Scaffold(
        bottomBar = {
            if (showBottomBar) {
                BottomNavigationBar(
                    navController = navController,
                    cartCount = cartCount
                )
            }
        }
    ) { paddingValues ->
        EcommerceNavGraph(
            navController = navController,
            modifier = Modifier.padding(paddingValues)
        )
    }
}

// Получение аргументов в ViewModel через SavedStateHandle
@HiltViewModel
class DetailViewModel @Inject constructor(
    savedStateHandle: SavedStateHandle,
    private val getProductByIdUseCase: GetProductByIdUseCase
) : ViewModel() {

    // Автоматически получаем productId из навигации
    private val productId: Int = savedStateHandle.get<Int>("productId") ?: 0

    init {
        loadProduct(productId)
    }
}
