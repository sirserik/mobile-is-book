VIPER архитектура

View, Interactor, Presenter, Entity, Router

Зачем нужна архитектура?

Без архитектуры код превращается в "спагетти" — всё в одном месте, сложно тестировать и поддерживать. VIPER разделяет ответственность между компонентами.

V
View
I
Interactor
P
Presenter
E
Entity
R
Router

Компоненты VIPER

View (UIViewController)

Отображает UI и передаёт действия пользователя в Presenter. Ничего не знает о бизнес-логике.

Кнопка нажата → сообщаем Presenter

Presenter

Связующее звено. Получает данные от Interactor, форматирует и передаёт во View.

View ↔ Presenter ↔ Interactor

Interactor

Бизнес-логика: API запросы, работа с базой данных, валидация. Работает с Entity.

Загрузить товары → отправить в Presenter

Entity

Модели данных. Простые структуры без логики.

struct Product { name, price, image }

Router

Навигация между экранами. Создаёт модули и управляет переходами.

Открыть экран деталей товара

Протоколы — контракты

Компоненты общаются через протоколы. Это позволяет легко тестировать и заменять реализации.

ProductsProtocols.swift
// MARK: - View Input (Presenter → View)
protocol ProductsViewInput: AnyObject {
    func showProducts(_ products: [ProductViewModel])
    func showLoading()
    func hideLoading()
    func showError(_ message: String)
}

// MARK: - View Output (View → Presenter)
protocol ProductsViewOutput: AnyObject {
    func viewDidLoad()
    func didSelectProduct(at index: Int)
    func didTapAddToCart(at index: Int)
}

// MARK: - Interactor Input (Presenter → Interactor)
protocol ProductsInteractorInput: AnyObject {
    func fetchProducts()
    func addToCart(product: Product)
}

// MARK: - Interactor Output (Interactor → Presenter)
protocol ProductsInteractorOutput: AnyObject {
    func didFetchProducts(_ products: [Product])
    func didFailToFetchProducts(_ error: Error)
    func didAddToCart()
}

// MARK: - Router
protocol ProductsRouterInput: AnyObject {
    func showProductDetail(_ product: Product)
    func showCart()
}

class ProductsViewController: UIViewController {

    var presenter: ProductsViewOutput!

    private let tableView = UITableView()
    private var products: [ProductViewModel] = []

    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        presenter.viewDidLoad()  // Сообщаем Presenter
    }
}

extension ProductsViewController: ProductsViewInput {

    func showProducts(_ products: [ProductViewModel]) {
        self.products = products
        tableView.reloadData()
    }

    func showLoading() {
        // Показать индикатор загрузки
    }

    func hideLoading() {
        // Скрыть индикатор загрузки
    }

    func showError(_ message: String) {
        // Показать alert
    }
}

extension ProductsViewController: UITableViewDelegate {
    func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
        presenter.didSelectProduct(at: indexPath.row)

ProductsPresenter.swift
class ProductsPresenter {

    weak var view: ProductsViewInput?
    var interactor: ProductsInteractorInput!
    var router: ProductsRouterInput!

    private var products: [Product] = []
}

extension ProductsPresenter: ProductsViewOutput {

    func viewDidLoad() {
        view?.showLoading()
        interactor.fetchProducts()
    }

    func didSelectProduct(at index: Int) {
        let product = products[index]
        router.showProductDetail(product)
    }

    func didTapAddToCart(at index: Int) {
        let product = products[index]
        interactor.addToCart(product: product)
    }
}

extension ProductsPresenter: ProductsInteractorOutput {

    func didFetchProducts(_ products: [Product]) {
        self.products = products
        view?.hideLoading()

        // Преобразуем Entity → ViewModel
        let viewModels = products.map { product in
            ProductViewModel(
                name: product.name,
                price: formatPrice(product.price),
                imageURL: product.imageURL
            )
        }
        view?.showProducts(viewModels)
    }

    func didFailToFetchProducts(_ error: Error) {
        view?.hideLoading()
        view?.showError(error.localizedDescription)
    }

    func didAddToCart() {
        // Показать уведомление
    }

    private func formatPrice(_ price: Double) -> String {
        return String(format: "%.0f ₸", price)
    }
}
ProductsInteractor.swift
class ProductsInteractor {

    weak var presenter: ProductsInteractorOutput?
    private let apiService: APIServiceProtocol
    private let cartService: CartServiceProtocol

    init(apiService: APIServiceProtocol, cartService: CartServiceProtocol) {
        self.apiService = apiService
        self.cartService = cartService
    }
}

extension ProductsInteractor: ProductsInteractorInput {

    func fetchProducts() {
        Task {
            do {
                let products = try await apiService.fetchProducts()
                await MainActor.run {
                    presenter?.didFetchProducts(products)
                }
            } catch {
                await MainActor.run {
                    presenter?.didFailToFetchProducts(error)
                }
            }
        }
    }

    func addToCart(product: Product) {
        cartService.add(product)
        presenter?.didAddToCart()
    }
}
ProductsRouter.swift
class ProductsRouter {

    weak var viewController: UIViewController?

    // Фабричный метод — создаёт весь модуль
    static func createModule() -> UIViewController {
        let view = ProductsViewController()
        let presenter = ProductsPresenter()
        let interactor = ProductsInteractor(
            apiService: APIService(),
            cartService: CartService.shared
        )
        let router = ProductsRouter()

        // Связываем компоненты
        view.presenter = presenter
        presenter.view = view
        presenter.interactor = interactor
        presenter.router = router
        interactor.presenter = presenter
        router.viewController = view

        return view
    }
}

extension ProductsRouter: ProductsRouterInput {

    func showProductDetail(_ product: Product) {
        let detailVC = ProductDetailRouter.createModule(with: product)
        viewController?.navigationController?.pushViewController(detailVC, animated: true)
    }

    func showCart() {
        let cartVC = CartRouter.createModule()
        viewController?.present(cartVC, animated: true)
    }
}
